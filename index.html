/**
 * Given an array of objects, return an object whose keys are the values of a specified key in the
 * objects, and whose values are arrays of the objects that have that value for the specified key
 * @param arr - The array to be grouped.
 * @param key - The key to group by.
 */
const objgroupBy = (arr, key) => {
	return arr.reduce( (acc, item, i) => {
		if( Object.keys(item).includes( key ) ) {
			// acc[item[key]] = (item || [] ).concat(arr[i])
			/* Adding the item to the array of the key. */
			acc[item[key]] = ( acc[item[key]] || []).concat(arr[i])
		} else {
			console.log(item, 'does not have'+ key );
		}

		return acc;
	}, {})
}

const findMissingPerson = function (params) {
	let n = params.length + 1;
	let sum = 0;
	for (let a in params) {
		sum += params[a];
	}
	console.log(sum);
	return Math.floor( (n * (n + 1)) /2 ) - sum
}
